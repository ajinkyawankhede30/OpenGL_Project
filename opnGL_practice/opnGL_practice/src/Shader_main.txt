#shader vertex
#version 430 core

layout(location = 0) in vec3 positions;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;


void main()
{
	gl_Position = projection * view * model * vec4(positions, 1.0f);

};



#shader fragment
#version 430 core

out vec4 color;
uniform vec3 objectColor;
uniform vec3 lightColor;


void main()
{
	color = vec4(lightColor * objectColor, 1.0f);
};
