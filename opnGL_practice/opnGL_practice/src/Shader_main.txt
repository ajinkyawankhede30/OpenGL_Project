#shader vertex
#version 330 core

layout(location = 0) in vec4 positions;
layout(location = 1) in vec3 vertex_colors;
out vec3 out_colors;
void main()
{
	gl_Position = positions;
	out_colors = vertex_colors;
};



#shader fragment
#version 330 core

out vec4 color;
in vec3 out_colors;
void main()
{
	color = vec4(out_colors,1.0f);
};
