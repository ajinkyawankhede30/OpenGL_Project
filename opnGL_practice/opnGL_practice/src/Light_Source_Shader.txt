#shader vertex
#version 430 core

layout(location = 0)in vec3 positions;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

void main()
{
	gl_Position = model * view * projection * vec4(positions,1.0);
};


#shader fragment
#version 430 core

out vec4 color;

void main()
{
	color = vec4(1.0);
};
